vec2 get_velocity(vec2 p) {
  vec2 v = vec2(0., 0.);
  float c,dot_c,f;
  c = p.x;
  dot_c = p.y;

  if (c<-1.0 || (c<1.0 && dot_c>0.0))
  {
    f = dot_c - 1.0;
  }
  else
  {
    f = dot_c + 1.0;
  }

  v.x = p.y;
  v.y = -f;
  
  return v;
}
